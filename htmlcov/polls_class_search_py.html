279def generer_liste_reponse(produit, liste_reponse, user_current, compare): 

280 ''' vérifie si un produit trouvé n'est pas déjà enregistré dans les favoris 

281 checks if a product found is not already saved in favorites ''' 

282 save = True 

283 try: 

284 user_favoris = Favoris.objects.filter(user__exact=user_current.id) 

285 for favoris in user_favoris: 

286 if favoris.produits.id == compare.id: 

287 save = False 

288 break 

289 else: 

290 save = True 

291 produit['favoris'] = save 

292 liste_reponse.append(produit) 

293 except BaseException: 

294 produit['favoris'] = save 

295 liste_reponse.append(produit) 

296 nw_liste_reponse = sorted(liste_reponse, key=lambda k: k['grade']) 

297 return nw_liste_reponse
